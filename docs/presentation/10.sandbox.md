9.1 Debugging Containers Like a Pro

```sh
docker compose run sandbox sh

apk update && apk add curl tar

sleep 3600 &
ps -ef

go build -o cfs .
./cfs pull alpine
./cfs run alpine /bin/sh

ps -ef
touch hi.txt


docker run -it --privileged --entrypoint sh ghcr.io/mohammadne/cfs:v0.0.4
cfs pull alpine
cfs run alpine /bin/sh
```
